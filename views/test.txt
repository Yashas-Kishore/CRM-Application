  <script>
    let taskModal = document.getElementById('taskModal');
    let taskDateElement = document.getElementById('taskDate');
    let currentTaskDate = null;

    function openTaskModal(date) {
      currentTaskDate = date;
      taskDateElement.innerText = `Date: ${date}`;
      taskModal.classList.add('open');
    }

    function saveTask() {
      // Logic to save the task, likely an API request or local storage
      const taskTitle = document.getElementById('taskTitle').value;
      const taskDetails = document.getElementById('taskDetails').value;
      console.log(`Saving task for ${currentTaskDate}: ${taskTitle} - ${taskDetails}`);
      // Close modal after saving
      taskModal.classList.remove('open');
    }

    function prevMonth() {
      // Logic to navigate to previous month

    }

      // Logic to navigate to next month
      window.location.href = `/calendar?month=next`;
